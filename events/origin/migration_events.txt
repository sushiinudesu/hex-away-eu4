namespace = hex_origin_migration_event

country_event = {
	id = hex_origin_migration_event.1
	title = hex_origin_migration_event.1.t
	desc = hex_origin_migration_event.1.d
	
	picture = HEX_EVENT_PICTURE_WIZARD_AWAKE

	is_triggered_only = yes

	option = {
		name = hex_origin_migration_event.1.a

		hidden_effect = {
			random_province = {
				limit = {
					province_group = hex_wizard_migration_province_group
					NOT = { has_province_modifier = hex_wizard_base }
				}
				province_event = { id = hex_origin_migration_event.2 }
			}
		}
	}

	after = {
		hidden_effect = {
			remove_province_modifier = hex_wizard_base
		}
	}
}
