country_event = {
	id = hex_origin_wizard_events.1
	title = hex_origin_wizard_events.1.t
	desc = hex_origin_wizard_events.1.d
	
	picture = HEX_EVENT_PICTURE_WIZARD_AWAKE

	fire_only_once = yes

	mean_time_to_happen = {
		days = 730
	}

	goto = 21

	option = {
		name = hex_origin_wizard_events.1.a

		hidden_effect = {
			random_list = {
				1 = { country_event = { id = hex_origin_wizard_events.7 days = 100 random = 1725 } } 
				1 = { country_event = { id = hex_origin_wizard_events.8 days = 100 random = 1725 } } 
			}
		}

		ai_chance = { factor = 1 }
	}

	after = {
		hidden_effect = {
			21 = {
				if = { limit = { is_city = no } hex_add_to_city = yes }
				add_permanent_province_modifier = { name = hex_wizard_base duration = -1 }
				province_event = { id = hex_origin_migration_events.1 days = 365 random = 365 }
			}
		}
	}
}
