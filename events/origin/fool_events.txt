	fire_only_once = yes

	mean_time_to_happen = {
		days = 730

		name = hex_origin_fool_events.1.a

	}

	after = {

				province_event = { id = hex_origin_migration_events.9 days = 365 random = 365 }
			}
		}
	}
}
