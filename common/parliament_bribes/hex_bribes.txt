hex_bribe_capital = {
	trigger = {
		FROM = { is_capital = yes }
		NOT = { has_country_flag = hex_questing }
	}
	
	effect = {
		hidden_effect = {
			set_country_flag = hex_questing

			FROM = {
				back_current_issue = no
				province_event = { id = hex_quest_events.2 }
			}
		}
	}

	chance = { factor = 1 }
	
	ai_will_do = { factor = 0 }
}

hex_bribe_non_capital = {
	trigger = {
		FROM = { NOT = { is_capital = yes } }
		NOT = { has_country_flag = hex_questing }
		hex_can_finish_quest = no
	}
	
	effect = {
		hidden_effect = {
			set_country_flag = hex_questing

			FROM = {
				back_current_issue = no
				province_event = { id = hex_quest_events.2 }
			}
		}
	}

	chance = { factor = 1 }
	
	ai_will_do = { factor = 0 }
}
