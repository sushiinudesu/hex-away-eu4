hex_set_familiar = {
	if = {
		limit = { hex_consorts_mechanic_available = yes }
		hex_set_familiar_$familiar$ = yes
	}
	else = {
		add_country_modifier = { name = hex_familiar_$familiar$ duration = -1 }
	}
}

hex_set_familiar_bear_cub = {
	define_consort = {
		name = "Bear"
		country_of_origin = ROOT
		dynasty = "'the Small Cub"
		culture = hex_nature_bear_culture
		adm = 1
		dip = 1
		mil = 1
	}
	add_queen_personality = hex_familiar_bear_personality
}
