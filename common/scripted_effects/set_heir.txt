hex_set_heir = {
	export_to_variable = { variable_name = hex_heir_adm value = heir_adm }
	export_to_variable = { variable_name = hex_heir_dip value = heir_dip }
	export_to_variable = { variable_name = hex_heir_mil value = heir_mil }

	hex_set_heir_$character$ = yes

	hex_set_heir_skills = yes
	add_heir_personality = hex_awakening_awake_personality
	swap_non_generic_missions = yes
}

hex_set_heir_skills = {
	hex_change_heir_skill = { skill = adm }
	hex_change_heir_skill = { skill = dip }
	hex_change_heir_skill = { skill = mil }
}

hex_change_heir_skill = {
	change_heir_$skill$ = -6

	if = { limit = { check_variable = { which = hex_heir_$skill$ value = 6 } } change_heir_$skill$ = 6 }
	else_if = { limit = { check_variable = { which = hex_heir_$skill$ value = 5 } } change_heir_$skill$ = 5 }
	else_if = { limit = { check_variable = { which = hex_heir_$skill$ value = 4 } } change_heir_$skill$ = 4 }
	else_if = { limit = { check_variable = { which = hex_heir_$skill$ value = 3 } } change_heir_$skill$ = 3 }
	else_if = { limit = { check_variable = { which = hex_heir_$skill$ value = 2 } } change_heir_$skill$ = 2 }
	else = { change_heir_$skill$ = 1 }
}
