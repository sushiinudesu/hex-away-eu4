hex_setup_country = {
	if = {
		limit = { NOT = { has_reform_on_level = 0 }}
		hex_setup_administration = yes
	}

	if = {
		limit = { NOT = { has_country_flag = hex_setup_finished } }

		add_adm_power = -10000
		add_dip_power = -10000
		add_mil_power = -10000
		add_war_exhaustion = -100

		hex_clean_treasury = yes
		add_treasury = 1

		hex_setup_vision = yes

		if = {
			limit = { ai = no }

			add_item = { awakening_robes = yes skill = 1 }
			add_item = { gold_coin = yes skill = 1 }

			hex_setup_kills = yes
		}

		if = { 
			limit = { hex_professionalism_mechanic_available = no }
			set_variable = { which = hex_military_professionalism value = 0 }
			add_country_modifier = { name = hex_military_professionalism_0 duration = -1 }
		}

		set_country_flag = hex_setup_finished
	}
}
