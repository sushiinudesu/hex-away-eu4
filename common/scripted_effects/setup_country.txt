hex_setup_country = {
	if = {
		limit = { NOT = { has_reform_on_level = 0 }}
		hex_setup_administration = yes
	}

	if = {
		limit = { NOT = { has_country_flag = hex_setup_finished } }

		add_adm_power = -10000
		add_dip_power = -10000
		add_mil_power = -10000
		add_war_exhaustion = -100

		hex_setup_vision = yes

		set_country_flag = hex_setup_finished
	}
}
