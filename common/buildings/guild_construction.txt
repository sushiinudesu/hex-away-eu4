hex_building_lion_fort = {
	cost = 0
	time = 6

	indestructible = yes
	one_per_country = yes

	build_trigger = { always = no }
	
	on_construction_started = {
		add_permanent_province_modifier = {
			name = hex_faction_has_construction
			duration = -1
		}
	}
	on_construction_canceled = {
		remove_province_modifier = hex_faction_has_construction
	}
	on_built = {
		remove_building = hex_building_lion_fort
		remove_province_modifier = hex_faction_has_construction
		set_province_flag = hex_lion_fort_flag
		hidden_effect = { province_event = { id = hex_city_projects_events.1 } }
	}

	ai_will_do = { factor = 100 }
}
